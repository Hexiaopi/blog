    string字符串

    hexiaopi 1/23/2022 Go基础

    # 声明

    关键字定义： string

    零值： ""

    var str string
    1

    # 操作

    字符串支持一些简单的运算或操作，如下：

    操作 含义 示例 结果
    x+y 字符串连接 "Go"+"lang" "Golang"
    len(x) 计算字符串长度 len("Golang") 6
    str[i] 取字符 "Golang"[3] a

    # 比较

    支持各种比较关系运算符: ==、!=、>=、<=、>、<

    package main

import "fmt"

func main() {
	// ==
	s1 := "世界和平"
	s2 := "世界" + "和平"
	fmt.Println(s1 == s2) // true

	// !=
	s1 = "Go"
	s2 = "C"
	fmt.Println(s1 != s2) // true

	// < and <=
	s1 = "12345"
	s2 = "23456"
	fmt.Println(s1 < s2)  // true
	fmt.Println(s1 <= s2) // true

	// > and >=
	s1 = "12345"
	s2 = "123"
	fmt.Println(s1 > s2)  // true
	fmt.Println(s1 >= s2) // true
}

    # 遍历

    Go语言支持UTF-8格式编码，因此字符串中字符可以是ASCII字符，也可以是Unicode字符

    • byte类型来表示ASCII字符
    • rune类型来表示Unicode字符
    str := "中国欢迎您"

    # 底层结构

    参见$GOROOT/src/runtime/string.go

    type stringStruct struct {
	str unsafe.Pointer
	len int
}
    1
    2
    3
    4
    • str是指针，指向底层数据结构，其实是[]byte
    • len是字符串长度，因此len(s)是O(1)复杂度的操作

    因此传参是可以将字符串作为参数，而不必传字符串指针。

    字符串的内容可以用类似数组下标的方式获取，例如str[0]，但与数组不同，字符串的内容初始化后不可修改注意，只能重新构造新的字符串，带来的好处是线程安全

    package main

import "fmt"

func main() {
	var s string = "hello"
	s1 := []byte(s)
	s1[0] = 'z'
	fmt.Println(s, string(s1)) //hello zello
}

    # 字符串高效构造方式

    Go语言构造字符串的方式有：

    • +
    • fmt.Sprintf
    • strings.Join
    • strings.Build
    • bytes.Buffer

    但哪种方法最为高效呢？

    import (
	"bytes"
	"fmt"
	"strings"
	"testing"
)

var sl []string = []string{
	"Rob Pike ",
	"Robert Griesemer ",
	"Ken Thompson ",
}

func concatStringByOperator(sl []string) string {
	var s string
	for _, v := range sl {
		s += v
	}
	return s
}

func concatStringBySprintf(sl []string) string {
	var s string
	for _, v := range sl {
		s = fmt.Sprintf("%s%s", s, v)
	}
	return s
}

func concatStringByJoin(sl []string) string {
	return strings.Join(sl, "")
}

func concatStringByStringsBuilder(sl []string) string {
	var b strings.Builder
	for _, v := range sl {
		b.WriteString(v)
	}
	return b.String()
}

func concatStringByStringsBuilderWithInitSize(sl []string) string {
	var b strings.Builder
	b.Grow(64)
	for _, v := range sl {
		b.WriteString(v)
	}
	return b.String()
}

func concatStringByBytesBuffer(sl []string) string {
	var b bytes.Buffer
	for _, v := range sl {
		b.WriteString(v)
	}
	return b.String()
}

func concatStringByBytesBufferWithInitSize(sl []string) string {
	buf := make([]byte, 0, 64)
	b := bytes.NewBuffer(buf)
	for _, v := range sl {
		b.WriteString(v)
	}
	return b.String()
}
